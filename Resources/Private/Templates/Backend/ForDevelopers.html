<?php

declare(strict_types=1);

namespace Vendor\MyExtension\Service;

use NITSAN\NsT3AF\Api\AiOptions;
use NITSAN\NsT3AF\Api\AiServiceInterface;

final class MyFeatureService
{
    public function __construct(private readonly AiServiceInterface $ai) {}

    public function summarize(string $input): string
    {
        $response = $this->ai->complete(
            $input,
            new AiOptions(
                // Replace namespace, extensionKey, featureKey, and requestSource for your extension.
                extensionKey: 'my_extension',
                featureKey: 'summary',
                requestSource: 'backend_module',
            ),
        );

        return $response->content;
    }
}
